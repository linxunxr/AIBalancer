# 智衡(Balancer) - 账户管理系统设计方案

## 一、账户管理功能架构

### 1.1 功能概述
账户管理系统用于管理多个AI平台的API账户，包括DeepSeek、OpenAI、Anthropic等，提供统一的API密钥管理、余额监控、使用统计等功能。

### 1.2 功能模块
```
账户管理模块包含：
1. 账户列表 - 查看所有平台账户状态
2. 账户详情 - 查看单个账户详细信息
3. 添加账户 - 添加新的平台账户
4. 编辑账户 - 修改账户信息
5. 删除账户 - 移除账户
6. API测试 - 测试API连接状态
7. 批量操作 - 批量刷新、导出等
8. 使用统计 - 账户使用情况分析
```

```typescript
// 实体类实现
export class AccountEntity implements Account {
  constructor(
    public id: string,
    public name: string,
    public type: AccountType,
    public status: AccountStatus,
    public apiKeys: ApiKeyInfo[],
    public currentBalance: number,
    public currency: string,
    public usage: AccountUsage,
    public alerts: AccountAlert[],
    public metadata: {
      organization?: string
      project?: string
      tags: string[]
      notes?: string
      customFields: Record<string, any>
    },
    public settings: {
      autoRefresh: boolean
      refreshInterval: number
      lowBalanceThreshold: number
      enableAlerts: boolean
      usageLimit?: number
    },
    public createdAt: Date,
    public updatedAt: Date,
    public lastSyncedAt: Date
  ) {}

  static create(params: Omit<Account, 'id' | 'createdAt' | 'updatedAt' | 'lastSyncedAt'>): AccountEntity {
    const now = new Date()
    return new AccountEntity(
      crypto.randomUUID(),
      params.name,
      params.type,
      params.status,
      params.apiKeys,
      params.currentBalance,
      params.currency,
      params.usage,
      params.alerts,
      params.metadata,
      params.settings,
      now,
      now,
      now
    )
  }

  static createFromRequest(request: AccountCreateRequest): AccountEntity {
    const apiKeyInfo: ApiKeyInfo = {
      id: crypto.randomUUID(),
      key: '', // 这里应该加密存储
      lastUsed: new Date(),
      usageCount: 0,
      isActive: true,
      createdAt: new Date()
    }
  
    const defaultUsage: AccountUsage = {
      totalTokens: 0,
      promptTokens: 0,
      completionTokens: 0,
      totalCost: 0,
      lastUsed: new Date(),
      dailyAverage: 0,
      monthlyUsage: []
    }
  
    const defaultSettings = {
      autoRefresh: true,
      refreshInterval: 30,
      lowBalanceThreshold: 50,
      enableAlerts: true
    }
  
    return AccountEntity.create({
      name: request.name,
      type: request.type,
      status: AccountStatus.TESTING,
      apiKeys: [apiKeyInfo],
      currentBalance: 0,
      currency: 'CNY',
      usage: defaultUsage,
      alerts: [],
      metadata: {
        organization: request.organization,
        project: request.project,
        tags: request.tags || [],
        notes: request.notes,
        customFields: {}
      },
      settings: { ...defaultSettings, ...request.settings },
      createdAt: new Date(),
      updatedAt: new Date(),
      lastSyncedAt: new Date()
    })
  }

  update(params: AccountUpdateRequest): void {
    if (params.name) this.name = params.name
    if (params.status) this.status = params.status
    if (params.apiKeys) this.apiKeys = params.apiKeys
    if (params.metadata) {
      this.metadata = { ...this.metadata, ...params.metadata }
    }
    if (params.settings) {
      this.settings = { ...this.settings, ...params.settings }
    }
    this.updatedAt = new Date()
  }

  addApiKey(key: string): void {
    const apiKeyInfo: ApiKeyInfo = {
      id: crypto.randomUUID(),
      key: '', // 加密存储
      lastUsed: new Date(),
      usageCount: 0,
      isActive: true,
      createdAt: new Date()
    }
    this.apiKeys.push(apiKeyInfo)
    this.updatedAt = new Date()
  }

  deactivateApiKey(keyId: string): void {
    const key = this.apiKeys.find(k => k.id === keyId)
    if (key) {
      key.isActive = false
      this.updatedAt = new Date()
    }
  }

  recordUsage(tokens: number, cost: number, type: 'prompt' | 'completion' = 'prompt'): void {
    this.usage.totalTokens += tokens
    this.usage.totalCost += cost
  
    if (type === 'prompt') {
      this.usage.promptTokens += tokens
    } else {
      this.usage.completionTokens += tokens
    }
  
    this.usage.lastUsed = new Date()
  
    // 更新月使用量
    const currentMonth = new Date().toISOString().slice(0, 7)
    const monthUsage = this.usage.monthlyUsage.find(m => m.month === currentMonth)
  
    if (monthUsage) {
      monthUsage.tokens += tokens
      monthUsage.cost += cost
    } else {
      this.usage.monthlyUsage.push({
        month: currentMonth,
        tokens,
        cost
      })
    }
  
    this.updatedAt = new Date()
  }

  addAlert(alert: Omit<AccountAlert, 'id' | 'triggeredAt' | 'resolved'>): void {
    const newAlert: AccountAlert = {
      id: crypto.randomUUID(),
      ...alert,
      triggeredAt: new Date(),
      resolved: false
    }
    this.alerts.push(newAlert)
    this.updatedAt = new Date()
  }

  resolveAlert(alertId: string): void {
    const alert = this.alerts.find(a => a.id === alertId)
    if (alert && !alert.resolved) {
      alert.resolved = true
      alert.resolvedAt = new Date()
      this.updatedAt = new Date()
    }
  }

  getFormattedBalance(): string {
    return `${this.currency} ${this.currentBalance.toFixed(2)}`
  }

  getUsagePercentage(limit?: number): number {
    const usageLimit = limit || this.settings.usageLimit
    if (!usageLimit) return 0
    return (this.usage.totalTokens / usageLimit) * 100
  }

  getDaysSinceLastUsed(): number {
    const diff = new Date().getTime() - this.usage.lastUsed.getTime()
    return Math.floor(diff / (1000 * 60 * 60 * 24))
  }

  shouldAlert(): boolean {
    if (!this.settings.enableAlerts) return false
  
    // 检查低余额
    if (this.currentBalance < this.settings.lowBalanceThreshold) {
      return true
    }
  
    // 检查使用量限制
    if (this.settings.usageLimit && this.usage.totalTokens > this.settings.usageLimit * 0.9) {
      return true
    }
  
    // 检查API密钥过期
    const expiringKey = this.apiKeys.find(key => 
      key.expiresAt && key.expiresAt.getTime() - Date.now() < 7 * 24 * 60 * 60 * 1000
    )
    return !!expiringKey
  }
}
```

```typescript
// src/models/repositories/AccountRepository.ts
import { BaseRepository, RepositoryError } from './BaseRepository'
import { Account, AccountEntity, AccountType, AccountStatus } from '../entities/Account'
import { invoke } from '@tauri-apps/api/tauri'

export class AccountRepository extends BaseRepository<Account, string> {
  protected tableName = 'accounts'

  protected toEntity(row: any): Account {
    return new AccountEntity(
      row.id,
      row.name,
      row.type,
      row.status,
      JSON.parse(row.api_keys),
      row.current_balance,
      row.currency,
      JSON.parse(row.usage),
      JSON.parse(row.alerts),
      JSON.parse(row.metadata),
      JSON.parse(row.settings),
      new Date(row.created_at),
      new Date(row.updated_at),
      new Date(row.last_synced_at)
    )
  }

  protected fromEntity(entity: Account): Record<string, any> {
    return {
      id: entity.id,
      name: entity.name,
      type: entity.type,
      status: entity.status,
      api_keys: JSON.stringify(entity.apiKeys),
      current_balance: entity.currentBalance,
      currency: entity.currency,
      usage: JSON.stringify(entity.usage),
      alerts: JSON.stringify(entity.alerts),
      metadata: JSON.stringify(entity.metadata),
      settings: JSON.stringify(entity.settings),
      created_at: entity.createdAt.toISOString(),
      updated_at: entity.updatedAt.toISOString(),
      last_synced_at: entity.lastSyncedAt.toISOString()
    }
  }

  protected async executeQuery(sql: string, params?: any[]): Promise<any[]> {
    try {
      return await invoke<any[]>('execute_sql', { sql, params })
    } catch (error) {
      throw new RepositoryError(`SQL execution failed: ${sql}`, error)
    }
  }

  // 自定义查询方法
  async findByType(type: AccountType): Promise<Account[]> {
    return this.findAll({ type } as Partial<Account>)
  }

  async findByStatus(status: AccountStatus): Promise<Account[]> {
    return this.findAll({ status } as Partial<Account>)
  }

  async findActiveAccounts(): Promise<Account[]> {
    return this.findAll({ status: AccountStatus.ACTIVE } as Partial<Account>)
  }

  async searchAccounts(query: string): Promise<Account[]> {
    try {
      const results = await this.executeQuery(
        `SELECT * FROM ${this.tableName} 
         WHERE name LIKE ? 
         OR metadata LIKE ? 
         OR type LIKE ?`,
        [`%${query}%`, `%${query}%`, `%${query}%`]
      )
    
      return results.map(row => this.toEntity(row))
    } catch (error) {
      throw new RepositoryError(`Search failed for query: ${query}`, error)
    }
  }

  async updateBalance(accountId: string, newBalance: number): Promise<Account> {
    const account = await this.findById(accountId)
    if (!account) {
      throw new RepositoryError(`Account not found: ${accountId}`)
    }
  
    const updatedAccount = new AccountEntity(
      account.id,
      account.name,
      account.type,
      account.status,
      account.apiKeys,
      newBalance,
      account.currency,
      account.usage,
      account.alerts,
      account.metadata,
      account.settings,
      account.createdAt,
      new Date(),
      new Date()
    )
  
    return await this.update(accountId, updatedAccount)
  }

  async updateStatus(accountId: string, status: AccountStatus): Promise<Account> {
    const account = await this.findById(accountId)
    if (!account) {
      throw new RepositoryError(`Account not found: ${accountId}`)
    }
  
    const updatedAccount = new AccountEntity(
      account.id,
      account.name,
      account.type,
      status,
      account.apiKeys,
      account.currentBalance,
      account.currency,
      account.usage,
      account.alerts,
      account.metadata,
      account.settings,
      account.createdAt,
      new Date(),
      account.lastSyncedAt
    )
  
    return await this.update(accountId, updatedAccount)
  }

  async getAccountsSummary(): Promise<{
    total: number
    byType: Record<AccountType, number>
    byStatus: Record<AccountStatus, number>
    totalBalance: number
    totalUsage: number
  }> {
    try {
      const accounts = await this.findAll()
    
      const byType: Record<AccountType, number> = {} as any
      const byStatus: Record<AccountStatus, number> = {} as any
    
      let totalBalance = 0
      let totalUsage = 0
    
      accounts.forEach(account => {
        byType[account.type] = (byType[account.type] || 0) + 1
        byStatus[account.status] = (byStatus[account.status] || 0) + 1
        totalBalance += account.currentBalance
        totalUsage += account.usage.totalTokens
      })
    
      return {
        total: accounts.length,
        byType,
        byStatus,
        totalBalance,
        totalUsage
      }
    } catch (error) {
      throw new RepositoryError('Failed to get accounts summary', error)
    }
  }

  async getAccountsWithLowBalance(threshold?: number): Promise<Account[]> {
    try {
      const accounts = await this.findAll()
      const balanceThreshold = threshold || 50
    
      return accounts.filter(account => 
        account.currentBalance < balanceThreshold && 
        account.status === AccountStatus.ACTIVE
      )
    } catch (error) {
      throw new RepositoryError('Failed to get accounts with low balance', error)
    }
  }

  async getRecentlyUpdatedAccounts(limit: number = 10): Promise<Account[]> {
    try {
      const results = await this.executeQuery(
        `SELECT * FROM ${this.tableName} 
         ORDER BY updated_at DESC 
         LIMIT ?`,
        [limit]
      )
    
      return results.map(row => this.toEntity(row))
    } catch (error) {
      throw new RepositoryError('Failed to get recently updated accounts', error)
    }
  }
}
```

```typescript
// src/models/services/AccountService.ts
import { AccountRepository } from '../repositories/AccountRepository'
import { 
  Account, AccountEntity, AccountType, AccountStatus, 
  AccountCreateRequest, AccountUpdateRequest, AccountAlert 
} from '../entities/Account'
import { BalanceService } from './BalanceService'

export interface IAccountService {
  // 账户操作
  createAccount(request: AccountCreateRequest): Promise<Account>
  getAccount(id: string): Promise<Account>
  updateAccount(id: string, request: AccountUpdateRequest): Promise<Account>
  deleteAccount(id: string): Promise<void>
  listAccounts(filter?: AccountFilter): Promise<Account[]>

  // 账户状态管理
  testAccountConnection(id: string): Promise<ConnectionTestResult>
  refreshAccountBalance(id: string): Promise<Account>
  refreshAllBalances(): Promise<Account[]>
  activateAccount(id: string): Promise<Account>
  deactivateAccount(id: string): Promise<Account>

  // 批量操作
  batchUpdateStatus(accountIds: string[], status: AccountStatus): Promise<Account[]>
  batchRefreshBalances(accountIds: string[]): Promise<Account[]>
  batchDeleteAccounts(accountIds: string[]): Promise<void>

  // 统计和查询
  getAccountsSummary(): Promise<AccountsSummary>
  searchAccounts(query: string): Promise<Account[]>
  getAccountsWithAlerts(): Promise<Account[]>
  getAccountsByType(type: AccountType): Promise<Account[]>

  // API密钥管理
  addApiKey(accountId: string, apiKey: string): Promise<Account>
  rotateApiKey(accountId: string, oldKeyId: string, newKey: string): Promise<Account>
  deactivateApiKey(accountId: string, keyId: string): Promise<Account>

  // 使用记录
  recordUsage(accountId: string, tokens: number, cost: number): Promise<Account>
}

export interface AccountFilter {
  types?: AccountType[]
  statuses?: AccountStatus[]
  tags?: string[]
  search?: string
  minBalance?: number
  maxBalance?: number
  hasAlerts?: boolean
  limit?: number
  offset?: number
}

export interface ConnectionTestResult {
  success: boolean
  message: string
  latency?: number
  balance?: number
  details?: Record<string, any>
}

export interface AccountsSummary {
  total: number
  active: number
  inactive: number
  withErrors: number
  totalBalance: number
  totalUsage: number
  byType: Record<AccountType, number>
  byStatus: Record<AccountStatus, number>
  recentActivity: Array<{
    accountId: string
    accountName: string
    action: string
    timestamp: Date
  }>
}

export class AccountService implements IAccountService {
  private accountRepository: AccountRepository
  private balanceService: BalanceService

  constructor(
    accountRepository?: AccountRepository,
    balanceService?: BalanceService
  ) {
    this.accountRepository = accountRepository || new AccountRepository()
    this.balanceService = balanceService || new BalanceService()
  }

  async createAccount(request: AccountCreateRequest): Promise<Account> {
    try {
      // 验证API密钥
      const testResult = await this.testApiKey(request.type, request.apiKey)
      if (!testResult.success) {
        throw new ServiceError(`API密钥验证失败: ${testResult.message}`)
      }
    
      // 创建账户实体
      const account = AccountEntity.createFromRequest(request)
    
      // 设置初始余额
      if (testResult.balance !== undefined) {
        account.currentBalance = testResult.balance
        account.status = AccountStatus.ACTIVE
      }
    
      // 加密存储API密钥
      const encryptedKey = await this.encryptApiKey(request.apiKey)
      account.apiKeys[0].key = encryptedKey
    
      // 保存到数据库
      const savedAccount = await this.accountRepository.create(account)
    
      // 记录创建事件
      await this.logAccountEvent(savedAccount.id, 'account_created', {
        type: request.type,
        name: request.name
      })
    
      return savedAccount
    } catch (error) {
      throw new ServiceError('创建账户失败', error)
    }
  }

  async getAccount(id: string): Promise<Account> {
    try {
      const account = await this.accountRepository.findById(id)
      if (!account) {
        throw new ServiceError(`账户不存在: ${id}`)
      }
      return account
    } catch (error) {
      throw new ServiceError('获取账户信息失败', error)
    }
  }

  async updateAccount(id: string, request: AccountUpdateRequest): Promise<Account> {
    try {
      const account = await this.getAccount(id)
    
      // 创建更新后的账户实体
      const updatedAccount = new AccountEntity(
        account.id,
        request.name || account.name,
        account.type,
        request.status || account.status,
        request.apiKeys || account.apiKeys,
        account.currentBalance,
        account.currency,
        account.usage,
        account.alerts,
        { ...account.metadata, ...request.metadata },
        { ...account.settings, ...request.settings },
        account.createdAt,
        new Date(),
        account.lastSyncedAt
      )
    
      const result = await this.accountRepository.update(id, updatedAccount)
    
      // 记录更新事件
      await this.logAccountEvent(id, 'account_updated', {
        updatedFields: Object.keys(request)
      })
    
      return result
    } catch (error) {
      throw new ServiceError('更新账户失败', error)
    }
  }

  async deleteAccount(id: string): Promise<void> {
    try {
      const deleted = await this.accountRepository.delete(id)
      if (!deleted) {
        throw new ServiceError(`删除账户失败: ${id}`)
      }
    
      // 记录删除事件
      await this.logAccountEvent(id, 'account_deleted', {})
    } catch (error) {
      throw new ServiceError('删除账户失败', error)
    }
  }

  async listAccounts(filter?: AccountFilter): Promise<Account[]> {
    try {
      let accounts = await this.accountRepository.findAll()
    
      // 应用过滤器
      if (filter) {
        if (filter.types && filter.types.length > 0) {
          accounts = accounts.filter(account => filter.types!.includes(account.type))
        }
      
        if (filter.statuses && filter.statuses.length > 0) {
          accounts = accounts.filter(account => filter.statuses!.includes(account.status))
        }
      
        if (filter.tags && filter.tags.length > 0) {
          accounts = accounts.filter(account => 
            filter.tags!.every(tag => account.metadata.tags.includes(tag))
          )
        }
      
        if (filter.search) {
          const searchLower = filter.search.toLowerCase()
          accounts = accounts.filter(account => 
            account.name.toLowerCase().includes(searchLower) ||
            account.metadata.notes?.toLowerCase().includes(searchLower) ||
            account.metadata.tags.some(tag => tag.toLowerCase().includes(searchLower))
          )
        }
      
        if (filter.minBalance !== undefined) {
          accounts = accounts.filter(account => account.currentBalance >= filter.minBalance!)
        }
      
        if (filter.maxBalance !== undefined) {
          accounts = accounts.filter(account => account.currentBalance <= filter.maxBalance!)
        }
      
        if (filter.hasAlerts !== undefined) {
          accounts = accounts.filter(account => 
            filter.hasAlerts ? account.alerts.length > 0 : account.alerts.length === 0
          )
        }
      
        // 应用分页
        if (filter.limit !== undefined) {
          const offset = filter.offset || 0
          accounts = accounts.slice(offset, offset + filter.limit)
        }
      }
    
      return accounts
    } catch (error) {
      throw new ServiceError('获取账户列表失败', error)
    }
  }

  async testAccountConnection(id: string): Promise<ConnectionTestResult> {
    try {
      const account = await this.getAccount(id)
      const apiKey = await this.decryptApiKey(account.apiKeys[0].key)
    
      const testResult = await this.testApiKey(account.type, apiKey)
    
      // 更新账户状态
      if (testResult.success) {
        await this.accountRepository.updateStatus(id, AccountStatus.ACTIVE)
      } else {
        await this.accountRepository.updateStatus(id, AccountStatus.ERROR)
      
        // 添加错误告警
        const alert: Omit<AccountAlert, 'id' | 'triggeredAt' | 'resolved'> = {
          type: 'api_error',
          message: `API连接测试失败: ${testResult.message}`,
          severity: 'error'
        }
        // 这里需要更新账户的alerts字段
      }
    
      return testResult
    } catch (error) {
      throw new ServiceError('测试账户连接失败', error)
    }
  }

  async refreshAccountBalance(id: string): Promise<Account> {
    try {
      const account = await this.getAccount(id)
      const apiKey = await this.decryptApiKey(account.apiKeys[0].key)
    
      // 调用对应平台的API获取余额
      const balance = await this.fetchBalanceFromApi(account.type, apiKey)
    
      // 更新账户余额
      const updatedAccount = await this.accountRepository.updateBalance(id, balance)
    
      // 检查是否需要告警
      await this.checkBalanceAlerts(updatedAccount)
    
      // 更新同步时间
      await this.accountRepository.update(id, {
        ...updatedAccount,
        lastSyncedAt: new Date()
      } as any)
    
      // 记录同步事件
      await this.logAccountEvent(id, 'balance_refreshed', {
        oldBalance: account.currentBalance,
        newBalance: balance
      })
    
      return updatedAccount
    } catch (error) {
      // 更新账户状态为错误
      await this.accountRepository.updateStatus(id, AccountStatus.ERROR)
      throw new ServiceError('刷新账户余额失败', error)
    }
  }

  async refreshAllBalances(): Promise<Account[]> {
    try {
      const accounts = await this.accountRepository.findActiveAccounts()
      const refreshPromises = accounts.map(account => 
        this.refreshAccountBalance(account.id).catch(error => {
          console.error(`刷新账户 ${account.name} 余额失败:`, error)
          return null
        })
      )
    
      const results = await Promise.all(refreshPromises)
```

```typescript
// src/viewmodels/accounts/useAccountListViewModel.ts
import { ref, reactive, computed, onMounted, onUnmounted } from 'vue'
import { BaseViewModel } from '../BaseViewModel'
import { accountService, AccountService } from '../../models/services/AccountService'
import { 
  Account, AccountType, AccountStatus, AccountFilter 
} from '../../models/entities/Account'

interface AccountListState {
  accounts: Account[]
  filteredAccounts: Account[]
  isLoading: boolean
  error: string | null
  filter: AccountFilter
  selectedAccounts: Set<string>
  totalCount: number
  page: number
  pageSize: number
  sortBy: string
  sortOrder: 'asc' | 'desc'

export class AccountListViewModel extends BaseViewModel<AccountListState> {
  private refreshInterval?: number

  constructor() {
    const initialState: AccountListState = {
      accounts: [],
      filteredAccounts: [],
      isLoading: false,
      error: null,
      filter: {
        types: [],
        statuses: [],
        tags: [],
        search: '',
        limit: 50,
        offset: 0
      },
      selectedAccounts: new Set(),
      totalCount: 0,
      page: 1,
      pageSize: 50,
      sortBy: 'updatedAt',
      sortOrder: 'desc'
    }
  
    super(initialState)
  }

  protected async onInitialize(): Promise<void> {
    await this.loadAccounts()
    this.setupAutoRefresh()
  }

  protected async onDispose(): Promise<void> {
    if (this.refreshInterval) {
      clearInterval(this.refreshInterval)
    }
  }

  // 公共方法
  public async refresh(): Promise<void> {
    await this.withLoading(async () => {
      await this.loadAccounts()
    })
  }

  public async applyFilter(newFilter: Partial<AccountFilter>): Promise<void> {
    this.state.filter = {
      ...this.state.filter,
      ...newFilter,
      limit: this.state.pageSize,
      offset: (this.state.page - 1) * this.state.pageSize
    }
  
    await this.loadAccounts()
    this.applyLocalFiltering()
  }

  public async changePage(newPage: number): Promise<void> {
    this.state.page = newPage
    this.state.filter.offset = (newPage - 1) * this.state.pageSize
  
    await this.loadAccounts()
  }

  public async changePageSize(newSize: number): Promise<void> {
    this.state.pageSize = newSize
    this.state.page = 1
    this.state.filter.limit = newSize
    this.state.filter.offset = 0
  
    await this.loadAccounts()
  }

  public async changeSort(sortBy: string, sortOrder: 'asc' | 'desc'): Promise<void> {
    this.state.sortBy = sortBy
    this.state.sortOrder = sortOrder
  
    this.applyLocalFiltering()
  }

  public toggleAccountSelection(accountId: string): void {
    if (this.state.selectedAccounts.has(accountId)) {
      this.state.selectedAccounts.delete(accountId)
    } else {
      this.state.selectedAccounts.add(accountId)
    }
  }

  public selectAllAccounts(): void {
    this.state.selectedAccounts = new Set(this.state.filteredAccounts.map(a => a.id))
  }

  public clearSelection(): void {
    this.state.selectedAccounts.clear()
  }

  public async deleteSelectedAccounts(): Promise<void> {
    const accountIds = Array.from(this.state.selectedAccounts)
    if (accountIds.length === 0) return
  
    try {
      this.state.isLoading = true
      await accountService.batchDeleteAccounts(accountIds)
    
      // 重新加载数据
      await this.loadAccounts()
      this.state.selectedAccounts.clear()
    } catch (error) {
      this.state.error = error instanceof Error ? error.message : '删除失败'
      throw error
    } finally {
      this.state.isLoading = false
    }
  }

  public async refreshSelectedBalances(): Promise<void> {
    const accountIds = Array.from(this.state.selectedAccounts)
    if (accountIds.length === 0) return
  
    try {
      this.state.isLoading = true
      await accountService.batchRefreshBalances(accountIds)
    
      // 重新加载数据
      await this.loadAccounts()

      this.state.isLoading = false

      this.state.isLoading = false
    }

  public getAccountStats() {
    const stats = {
      total: this.state.accounts.length,
      active: 0,
      inactive: 0,
      error: 0,
      totalBalance: 0,
      totalUsage: 0,
      byType: {} as Record<AccountType, number>,
      byStatus: {} as Record<AccountStatus, number>
    }
  
    this.state.accounts.forEach(account => {

      stats.byStatus[account.status] = (stats.byStatus[account.status] || 0) + 1
    
      stats.totalBalance += account.currentBalance
      stats.totalUsage += account.usage.totalTokens
    
      switch (account.status) {
        case AccountStatus.ACTIVE:
          stats.active++
          break
        case AccountStatus.INACTIVE:
          stats.inactive++
          break
        case AccountStatus.ERROR:
          stats.error++
          break
      }
    })

  private async loadAccounts(): Promise<void> {
    try {
      this.state.isLoading = true
      this.state.error = null
    
      const accounts = await accountService.listAccounts(this.state.filter)
      this.state.accounts = accounts
      this.state.totalCount = accounts.length // 注意：这里应该是总记录数，需要后端支持
    
      this.applyLocalFiltering()
    } catch (error) {
      this.state.error = error instanceof Error ? error.message : '加载账户列表失败'
      console.error('Failed to load accounts:', error)
    } finally {
      this.state.isLoading = false
    }
  }

  private applyLocalFiltering(): void {
    let filtered = [...this.state.accounts]
  
    // 应用排序
    filtered.sort((a, b) => {
      let valueA: any, valueB: any
    
      switch (this.state.sortBy) {
        case 'name':
          valueA = a.name.toLowerCase()
          valueB = b.name.toLowerCase()
          break
        case 'balance':
          valueA = a.currentBalance
          valueB = b.currentBalance
          break
        case 'status':
          valueA = a.status
          valueB = b.status
          break
        case 'lastUsed':
          valueA = a.usage.lastUsed.getTime()
          valueB = b.usage.lastUsed.getTime()
          break
        case 'updatedAt':
          valueA = a.updatedAt.getTime()
          valueB = b.updatedAt.getTime()
          break
        default:
          valueA = a.updatedAt.getTime()
          valueB = b.updatedAt.getTime()
      }
    
      if (valueA < valueB) return this.state.sortOrder === 'asc' ? -1 : 1
      if (valueA > valueB) return this.state.sortOrder === 'asc' ? 1 : -1
      return 0
    })
  
    this.state.filteredAccounts = filtered
  }

  private setupAutoRefresh(): void {
    // 每5分钟自动刷新一次
    this.refreshInterval = setInterval(() => {
      this.loadAccounts()
    }, 5 * 60 * 1000) as unknown as number
  }

  // 计算属性

    return this.state.selectedAccounts.size
  }

  public get hasSelected(): boolean {
    return this.state.selectedAccounts.size > 0
  }

  public get allSelected(): boolean {
    return this.state.selectedAccounts.size === this.state.filteredAccounts.length && 
           this.state.filteredAccounts.length > 0
  }

  public get totalPages(): number {
    return Math.ceil(this.state.totalCount / this.state.pageSize)
  }

  public get hasPrevPage(): boolean {
    return this.state.page > 1
  }

  public get hasNextPage(): boolean {
    return this.state.page < this.totalPages
  }
}
```

```vue
// 方法
const applyQuickFilter = (filterKey: string) => {
  activeQuickFilter.value = filterKey

  let newFilter: Partial<any> = {}

  switch (filterKey) {
    case 'active':
      newFilter.statuses = [AccountStatus.ACTIVE]
      break
    case 'error':
      newFilter.statuses = [AccountStatus.ERROR]
      break
    case 'low-balance':
      newFilter.minBalance = 0
      newFilter.maxBalance = 50
      break
    case 'recent':
      newFilter.sortBy = 'lastUsed'
      newFilter.sortOrder = 'desc'
      break
    default:
      newFilter = {}
  }

  applyFilter(newFilter)
}

    newFilter.types = selectedTypes.value

    newFilter.statuses = selectedStatuses.value

  if (selectedTags.value.length > 0) {
    newFilter.tags = selectedTags.value

  if (minBalance.value !== null) {
    newFilter.minBalance = minBalance.value
  }

  if (maxBalance.value !== null) {
    newFilter.maxBalance = maxBalance.value
  }

  applyFilter(newFilter)
  showAdvancedFilter.value = false
}
```

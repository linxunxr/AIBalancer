# 智衡(Balancer) - 日志系统设计方案

基于MVVM架构和Tauri技术栈，设计一个完整的日志系统，支持日志轮转、压缩、按时间段导出、多日志等级和开发/生产模式。

    public module: string,
    public message: string,
    public details?: Record<string, any>,
    public stackTrace?: string,
    public threadId?: string,
    public correlationId?: string
  ) {}

    return new LogEntryEntity(
      crypto.randomUUID(),
      params.timestamp,
      params.level,
      params.source,
      params.module,
      params.message,
      params.details,
      params.stackTrace,
      params.threadId,
      params.correlationId
    )
  }

  toJSON(): Record<string, any> {
    return {
      id: this.id,
      timestamp: this.timestamp.toISOString(),
      level: this.level,
      source: this.source,
      module: this.module,
      message: this.message,
      details: this.details,
      stackTrace: this.stackTrace,
      threadId: this.threadId,
      correlationId: this.correlationId
    }
  }

  format(format: 'text' | 'json' | 'csv'): string {
    switch (format) {
      case 'text':
        return `[${this.timestamp.toISOString()}] [${this.level}] [${this.source}/${this.module}] ${this.message}`
      case 'json':
        return JSON.stringify(this.toJSON(), null, 2)
      case 'csv':
        return `"${this.timestamp.toISOString()}","${this.level}","${this.source}","${this.module}","${this.message.replace(/"/g, '""')}"`
      default:
        return this.message
    }
  }
}

    [LogSource.SYSTEM]: LogLevel.INFO,
    [LogSource.BALANCE]: LogLevel.DEBUG,
    [LogSource.API]: LogLevel.INFO,
    [LogSource.DATABASE]: LogLevel.WARN,
    [LogSource.UI]: LogLevel.INFO,
    [LogSource.UPDATE]: LogLevel.INFO,
    [LogSource.PLUGIN]: LogLevel.DEBUG
  }
}

export class LogConfigEntity {
  constructor(
    public config: LogSystemConfig = DefaultLogConfig
  ) {}

  update(newConfig: Partial<LogSystemConfig>): void {
    this.config = {
      ...this.config,
      ...newConfig,
      rotation: {
        ...this.config.rotation,
        ...(newConfig.rotation || {})
      },
      export: {
        ...this.config.export,
        ...(newConfig.export || {})
      },
      sources: {
        ...this.config.sources,
        ...(newConfig.sources || {})
      }
    }
  }

  getSourceLevel(source: LogSource): LogLevel {
    return this.config.sources[source] || this.config.defaultLevel
  }

  shouldLog(source: LogSource, level: LogLevel): boolean {
    const sourceLevel = this.getSourceLevel(source)
    const levelPriority = {
      [LogLevel.TRACE]: 0,
      [LogLevel.DEBUG]: 1,
      [LogLevel.INFO]: 2,
      [LogLevel.WARN]: 3,
      [LogLevel.ERROR]: 4,
      [LogLevel.FATAL]: 5
    }
  
    return levelPriority[level] >= levelPriority[sourceLevel]
  }

  isDevelopmentMode(): boolean {
    return this.config.mode === LogMode.DEVELOPMENT
  }

  isProductionMode(): boolean {
    return this.config.mode === LogMode.PRODUCTION
  }
}

// 日志管理器
pub struct LogManager {
    config: LogConfig,
    log_dir: PathBuf,
    file_appender: Option<NonBlocking>,
    _guard: Option<WorkerGuard>,
}

impl LogManager {
    pub async fn new(log_dir: impl AsRef<Path>) -> Result<Self, String> {
        let log_dir = log_dir.as_ref().to_path_buf();
      
        // 创建日志目录
        fs::create_dir_all(&log_dir).await
            .map_err(|e| format!("Failed to create log directory: {}", e))?;
      
        // 加载或创建默认配置
        let config = Self::load_config(&log_dir).await.unwrap_or_default();
      
        Ok(Self {
            config,
            log_dir,
            file_appender: None,
            _guard: None,
        })
    }
  
    pub async fn initialize(&mut self) -> Result<(), String> {
        let mut layers = Vec::new();
      
        // 控制台输出层
        if self.config.console_output {
            let console_layer = fmt::layer()
                .with_writer(std::io::stdout)
                .with_ansi(true)
                .with_filter(self.create_env_filter());
            layers.push(console_layer.boxed());
        }
      
        // 文件输出层
        if self.config.file_output {
            let file_appender = self.create_file_appender().await?;
            let (non_blocking, guard) = tracing_appender::non_blocking(file_appender);
          
            let file_layer = fmt::layer()
                .with_writer(non_blocking.clone())
                .with_ansi(false)
                .with_filter(self.create_env_filter());
            layers.push(file_layer.boxed());
          
            self.file_appender = Some(non_blocking);
            self._guard = Some(guard);
        }
      
        // 初始化日志系统
        registry().with(layers).init();
      
        // 启动清理任务
        self.start_cleanup_task();
      
        tracing::info!("Log system initialized in {:?} mode", self.config.mode);
        Ok(())
    }
  
    async fn create_file_appender(&self) -> Result<RollingFileAppender, String> {
        let rotation = if self.config.rotation.enabled {
            Rotation::daily()
        } else {
            Rotation::never()
        };
      
        let log_file = self.log_dir.join("balancer.log");
      
        // 创建滚动文件appender
        let appender = RollingFileAppender::builder()
            .rotation(rotation)
            .filename_prefix("balancer")
            .filename_suffix("log")
            .max_log_files(self.config.rotation.max_files)
            .build(log_file)
            .map_err(|e| format!("Failed to create file appender: {}", e))?;
      
        Ok(appender)
    }
  
    fn create_env_filter(&self) -> EnvFilter {
        let mut filter_strings = Vec::new();
      
        // 添加默认级别
        filter_strings.push(format!("{}={}", crate_name!(), self.config.default_level));
      
        // 添加各来源的级别配置
        for (source, level) in &self.config.sources {
            filter_strings.push(format!("{}={}", source, level));
        }
      
        // 开发模式开启更详细的日志
        if self.config.mode == "development" {
            filter_strings.push("tracing=warn".to_string());
            filter_strings.push("hyper=info".to_string());
            filter_strings.push("reqwest=info".to_string());
        }
      
        EnvFilter::new(filter_strings.join(","))
    }
  
    async fn load_config(log_dir: &Path) -> Option<LogConfig> {
        let config_path = log_dir.join("log_config.json");
      
        match fs::read_to_string(&config_path).await {
            Ok(content) => {
                serde_json::from_str(&content).ok()
            }
            Err(_) => None,
        }
    }
  
    async fn save_config(&self) -> Result<(), String> {
        let config_path = self.log_dir.join("log_config.json");
        let config_json = serde_json::to_string_pretty(&self.config)
            .map_err(|e| format!("Failed to serialize config: {}", e))?;
      
        fs::write(&config_path, config_json).await
            .map_err(|e| format!("Failed to save config: {}", e))?;
      
        Ok(())
    }
  
    fn start_cleanup_task(&self) {
        let log_dir = self.log_dir.clone();
        let retention_days = self.config.rotation.retention_days;
        let compression = self.config.rotation.compression;
      
        tokio::spawn(async move {
            let mut interval = tokio::time::interval(tokio::time::Duration::from_secs(3600)); // 每小时检查一次
          
            loop {
                interval.tick().await;
              
                if let Err(e) = Self::cleanup_old_logs(&log_dir, retention_days, compression).await {
                    tracing::error!("Failed to cleanup old logs: {}", e);
                }
            }
        });
    }
  
    async fn cleanup_old_logs(
        log_dir: &Path,
        retention_days: u32,
        compression_enabled: bool,
    ) -> Result<(), String> {
        let cutoff_time = SystemTime::now()
            .checked_sub(std::time::Duration::from_secs(retention_days as u64 * 24 * 3600))
            .unwrap_or(UNIX_EPOCH);
      
        let mut entries = fs::read_dir(log_dir).await
            .map_err(|e| format!("Failed to read log directory: {}", e))?;
      
        while let Some(entry) = entries.next_entry().await.map_err(|e| format!("{}", e))? {
            let path = entry.path();
          
            if path.is_file() {
                let metadata = fs::metadata(&path).await
                    .map_err(|e| format!("Failed to get metadata for {:?}: {}", path, e))?;
              
                if let Ok(modified) = metadata.modified() {
                    if modified < cutoff_time {
                        // 删除旧文件
                        fs::remove_file(&path).await
                            .map_err(|e| format!("Failed to remove old log file {:?}: {}", path, e))?;
                      
                        tracing::debug!("Removed old log file: {:?}", path);
                    }
                }
              
                // 压缩旧文件
                if compression_enabled && !path.extension().map_or(false, |ext| ext == "gz") {
                    if let Err(e) = Self::compress_log_file(&path).await {
                        tracing::warn!("Failed to compress log file {:?}: {}", path, e);
                    }
                }
            }
        }
      
        Ok(())
    }
  
    async fn compress_log_file(log_file: &Path) -> Result<(), String> {
        let compressed_path = log_file.with_extension("log.gz");
      
        // 读取原文件内容
        let content = fs::read(log_file).await
            .map_err(|e| format!("Failed to read log file for compression: {}", e))?;
      
        // 压缩内容
        let mut encoder = GzEncoder::new(Vec::new(), GzCompression::default());
        std::io::Write::write_all(&mut encoder, &content)
            .map_err(|e| format!("Failed to compress data: {}", e))?;
        let compressed_data = encoder.finish()
            .map_err(|e| format!("Failed to finish compression: {}", e))?;
      
        // 写入压缩文件
        fs::write(&compressed_path, compressed_data).await
            .map_err(|e| format!("Failed to write compressed file: {}", e))?;
      
        // 删除原文件
        fs::remove_file(log_file).await
            .map_err(|e| format!("Failed to remove original file after compression: {}", e))?;
      
        tracing::info!("Compressed log file: {:?} -> {:?}", log_file, compressed_path);
        Ok(())
    }
  
    // Tauri命令：获取日志配置

    pub async fn get_log_config(&self) -> Result<LogConfig, String> {

    }

    }

impl LogManager {
    async fn query_logs_internal(&self, filter: LogQueryFilter) -> Result<Vec<LogEntry>, String> {
        let mut logs = Vec::new();
      
        // 读取所有日志文件
        let log_files = self.get_log_files().await?;
      
        for file in log_files {
            let file_logs = self.read_log_file(&file, &filter).await?;
            logs.extend(file_logs);
        }
      
        // 应用过滤条件
        let mut filtered_logs = self.filter_logs(logs, &filter).await?;
      
        // 排序（最新的在前）
        filtered_logs.sort_by(|a, b| b.timestamp.cmp(&a.timestamp));
      
        // 应用分页
        if let Some(limit) = filter.limit {
            let offset = filter.offset.unwrap_or(0);
            let start = offset as usize;
            let end = std::cmp::min(start + limit as usize, filtered_logs.len());
            filtered_logs = filtered_logs[start..end].to_vec();
        }
      
        Ok(filtered_logs)
    }
  
    async fn get_log_files(&self) -> Result<Vec<PathBuf>, String> {
        let mut files = Vec::new();
      
        let mut entries = fs::read_dir(&self.log_dir).await
            .map_err(|e| format!("Failed to read log directory: {}", e))?;
      
        while let Some(entry) = entries.next_entry().await.map_err(|e| format!("{}", e))? {
            let path = entry.path();
          
            if path.is_file() {
                let filename = path.file_name().unwrap_or_default().to_string_lossy();
              
                // 匹配日志文件模式
                if filename.starts_with("balancer") && 
                   (filename.ends_with(".log") || filename.ends_with(".log.gz")) {
                    files.push(path);
                }
            }
        }
      
        // 按修改时间排序（最新的在前）
        files.sort_by(|a, b| {
            let meta_a = fs::metadata(a).await.ok();
            let meta_b = fs::metadata(b).await.ok();
          
            let time_a = meta_a.and_then(|m| m.modified().ok());
            let time_b = meta_b.and_then(|m| m.modified().ok());
          
            time_b.cmp(&time_a)
        });
      
        Ok(files)
    }
  
    async fn read_log_file(
        &self,
        file_path: &Path,
        filter: &LogQueryFilter,
    ) -> Result<Vec<LogEntry>, String> {
        let mut logs = Vec::new();
      
        // 检查文件是否需要被读取（基于时间过滤）
        if let Some(start_time) = &filter.start_time {
            let file_modified = fs::metadata(file_path).await
                .and_then(|m| m.modified())
                .map_err(|e| format!("Failed to get file metadata: {}", e))?;
          
            let file_modified_str = DateTime::<Local>::from(file_modified).to_rfc3339();
            if file_modified_str < *start_time {
                return Ok(logs); // 文件修改时间早于查询开始时间，跳过
            }
        }
      
        // 读取文件内容
        let content = if file_path.extension().map_or(false, |ext| ext == "gz") {
            // 解压gzip文件
            let compressed = fs::read(file_path).await
                .map_err(|e| format!("Failed to read compressed file: {}", e))?;
          
            let mut decoder = flate2::read::GzDecoder::new(&compressed[..]);
            let mut decompressed = String::new();
            std::io::Read::read_to_string(&mut decoder, &mut decompressed)
                .map_err(|e| format!("Failed to decompress file: {}", e))?;
          
            decompressed
        } else {
            fs::read_to_string(file_path).await
                .map_err(|e| format!("Failed to read log file: {}", e))?
        };
      
        // 解析日志行
        for line in content.lines() {
            if let Some(log_entry) = self.parse_log_line(line) {
                // 检查时间过滤
                if let Some(start_time) = &filter.start_time {
                    if log_entry.timestamp < *start_time {
                        continue;
                    }
                }
                if let Some(end_time) = &filter.end_time {
                    if log_entry.timestamp > *end_time {
                        continue;
                    }
                }
              
                logs.push(log_entry);
            }
        }
      
        Ok(logs)
    }
  
    fn parse_log_line(&self, line: &str) -> Option<LogEntry> {
        // 解析标准的tracing日志格式
        // 示例: [2024-01-15T14:30:00Z INFO system::balance] 余额已更新
        let re = regex::Regex::new(r"^\[(?P<timestamp>[^\]]+)\]\s+\[(?P<level>[^\]]+)\]\s+\[(?P<source>[^\]:]+)(::(?P<module>[^\]]+))?\]\s+(?P<message>.*)$")
            .ok()?;
      
        let caps = re.captures(line)?;
      
        Some(LogEntry {
            id: uuid::Uuid::new_v4().to_string(),
            timestamp: caps.name("timestamp")?.as_str().to_string(),
            level: caps.name("level")?.as_str().to_string(),
            source: caps.name("source")?.as_str().to_string(),
            module: caps.name("module").map(|m| m.as_str().to_string()).unwrap_or_default(),
            message: caps.name("message")?.as_str().to_string(),
            details: None,
            stack_trace: None,
            thread_id: None,
            correlation_id: None,
        })
    }
  
    async fn filter_logs(
        &self,
        logs: Vec<LogEntry>,
        filter: &LogQueryFilter,
    ) -> Result<Vec<LogEntry>, String> {
        let mut filtered = logs;
      
        // 按级别过滤
        if let Some(levels) = &filter.levels {
            filtered.retain(|log| levels.contains(&log.level));
        }
      
        // 按来源过滤
        if let Some(sources) = &filter.sources {
            filtered.retain(|log| sources.contains(&log.source));
        }
      
        // 按模块过滤
        if let Some(modules) = &filter.modules {
            filtered.retain(|log| modules.contains(&log.module));
        }
      
        // 按关键词过滤
        if let Some(keywords) = &filter.keywords {
            filtered.retain(|log| {
                keywords.iter().any(|keyword| 
                    log.message.to_lowercase().contains(&keyword.to_lowercase()) ||
                    log.module.to_lowercase().contains(&keyword.to_lowercase()) ||
                    log.source.to_lowercase().contains(&keyword.to_lowercase())
                )
            });
        }
      
        Ok(filtered)
    }
  
    async fn export_logs_internal(
        &self,
        filter: LogQueryFilter,
        format: &str,
        include_details: bool,
    ) -> Result<String, String> {
        let logs = self.query_logs_internal(filter).await?;
      
        match format {
            "json" => {
                let export_logs: Vec<serde_json::Value> = logs.into_iter().map(|log| {
                    let mut log_obj = serde_json::json!({
                        "timestamp": log.timestamp,
                        "level": log.level,
                        "source": log.source,
                        "module": log.module,
                        "message": log.message,
                    });
                  
                    if include_details {
                        if let Some(details) = log.details {
                            log_obj["details"] = details;
                        }
                        if let Some(stack_trace) = log.stack_trace {
                            log_obj["stack_trace"] = serde_json::Value::String(stack_trace);
                        }
                    }
                  
                    log_obj
                }).collect();
              
                serde_json::to_string_pretty(&export_logs)
                    .map_err(|e| format!("Failed to serialize to JSON: {}", e))
            }
            "csv" => {
                let mut csv_writer = csv::Writer::from_writer(Vec::new());
              
                // 写入标题行
                let headers = if include_details {
                    vec!["timestamp", "level", "source", "module", "message", "details", "stack_trace"]
                } else {
                    vec!["timestamp", "level", "source", "module", "message"]
                };
              
                csv_writer.write_record(headers)
                    .map_err(|e| format!("Failed to write CSV header: {}", e))?;
              
                // 写入数据行
                for log in logs {
                    let mut record = vec![
                        log.timestamp,
                        log.level,
                        log.source,
                        log.module,
                        log.message,
                    ];
                  
                    if include_details {
                        let details_str = log.details
                            .map(|d| serde_json::to_string(&d).unwrap_or_default())
                            .unwrap_or_default();
                        let stack_trace_str = log.stack_trace.unwrap_or_default();
                      
                        record.push(details_str);
                        record.push(stack_trace_str);
                    }
                  
                    csv_writer.write_record(&record)
                        .map_err(|e| format!("Failed to write CSV record: {}", e))?;
                }
              
                let csv_data = csv_writer.into_inner()
                    .map_err(|e| format!("Failed to get CSV data: {}", e))?;
              
                String::from_utf8(csv_data)
                    .map_err(|e| format!("Failed to convert CSV to string: {}", e))
            }
            "text" => {
                let text_logs: Vec<String> = logs.into_iter().map(|log| {
                    let mut line = format!(
                        "[{}] [{}] [{}::{}] {}",
                        log.timestamp, log.level, log.source, log.module, log.message
                    );
                  
                    if include_details {
                        if let Some(details) = log.details {
                            line.push_str(&format!("\n  Details: {}", serde_json::to_string(&details).unwrap_or_default()));
                        }
                        if let Some(stack_trace) = log.stack_trace {
                            line.push_str(&format!("\n  Stack Trace:\n{}", stack_trace));
                        }
                    }
                  
                    line
                }).collect();
              
                Ok(text_logs.join("\n"))
            }
            _ => Err(format!("Unsupported export format: {}", format)),
        }
    }
  
    async fn list_log_files_internal(&self) -> Result<Vec<LogFileInfo>, String> {
        let mut log_files = Vec::new();
      
        for entry in fs::read_dir(&self.log_dir).await
            .map_err(|e| format!("Failed to read log directory: {}", e))? {
            let entry = entry.map_err(|e| format!("Failed to read directory entry: {}", e))?;
            let path = entry.path();
          
            if path.is_file() {
                let filename = path.file_name()
                    .unwrap_or_default()
                    .to_string_lossy()
                    .to_string();
              
                if filename.starts_with("balancer") && 
                   (filename.ends_with(".log") || filename.ends_with(".log.gz")) {
                    let metadata = fs::metadata(&path).await
                        .map_err(|e| format!("Failed to get file metadata: {}", e))?;
                  
                    let created = metadata.created()
                        .or_else(|_| metadata.modified())
                        .map_err(|e| format!("Failed to get file time: {}", e))?;
                  
                    let modified = metadata.modified()
                        .map_err(|e| format!("Failed to get modified time: {}", e))?;
                  
                    let compressed = filename.ends_with(".gz");
                  
                    // 估算日志数量
                    let log_count = if compressed {
                        0 // 压缩文件不计算
                    } else {
                        self.estimate_log_count(&path).await.unwrap_or(0)
                    };
                  
                    log_files.push(LogFileInfo {
                        filename,
                        path: path.to_string_lossy().to_string(),
                        size: metadata.len(),
                        created: DateTime::<Local>::from(created).to_rfc3339(),
                        modified: DateTime::<Local>::from(modified).to_rfc3339(),
                        compressed,
                        log_count,
                        period: LogPeriod {
                            start: DateTime::<Local>::from(created).to_rfc3339(),
                            end: DateTime::<Local>::from(modified).to_rfc3339(),
                        },
                    });
                }
            }
        }
      
        // 按修改时间排序（最新的在前）
        log_files.sort_by(|a, b| b.modified.cmp(&a.modified));
      
        Ok(log_files)
    }
  
    async fn estimate_log_count(&self, file_path: &Path) -> Result<u64, String> {
        let content = fs::read_to_string(file_path).await
            .map_err(|e| format!("Failed to read file for log count estimation: {}", e))?;
      
        // 统计包含时间戳的行数（简单的日志行估计）
        let count = content.lines()
            .filter(|line| line.starts_with('['))
            .count() as u64;
      
        Ok(count)
    }
  
    async fn cleanup_logs_internal(
        &self,
        older_than_days: Option<u32>,
    ) -> Result<u32, String> {
        let retention_days = older_than_days.unwrap_or(self.config.rotation.retention_days);
      
        let cutoff_time = SystemTime::now()
            .checked_sub(std::time::Duration::from_secs(retention_days as u64 * 24 * 3600))
            .ok_or("Failed to calculate cutoff time")?;
      
        let mut deleted_count = 0;
      
        for entry in fs::read_dir(&self.log_dir).await
            .map_err(|e| format!("Failed to read log directory: {}", e))? {
            let entry = entry.map_err(|e| format!("Failed to read directory entry: {}", e))?;
            let path = entry.path();
          
            if path.is_file() {
                let metadata = fs::metadata(&path).await
                    .map_err(|e| format!("Failed to get file metadata: {}", e))?;
              
                if let Ok(modified) = metadata.modified() {
                    if modified < cutoff_time {
                        fs::remove_file(&path).await
                            .map_err(|e| format!("Failed to delete old log file: {}", e))?;
                      
                        deleted_count += 1;
                        tracing::info!("Deleted old log file: {:?}", path);
                    }
                }
            }
        }
      
        Ok(deleted_count)
    }
}

static LOG_MANAGER: once_cell::sync::OnceCell<LogManager> = once_cell::sync::OnceCell::new();

pub fn init_log_manager(log_dir: impl AsRef<Path>) -> Result<(), String> {
    let rt = tokio::runtime::Runtime::new()
        .map_err(|e| format!("Failed to create runtime: {}", e))?;
  
    let log_manager = rt.block_on(async {
        LogManager::new(log_dir).await
    })?;
  
    LOG_MANAGER.set(log_manager)
        .map_err(|_| "Log manager already initialized".to_string())?;
  
    Ok(())
}

pub fn get_log_manager() -> Result<&'static LogManager, String> {
    LOG_MANAGER.get().ok_or("Log manager not initialized".to_string())
}

import { ref, reactive, computed, onMounted, onUnmounted } from 'vue'
import { BaseViewModel } from '../BaseViewModel'
import { loggingService, LoggingService } from '../../models/services/LoggingService'
import { LogEntry, LogLevel, LogSource, LogQueryFilter } from '../../models/entities/LogEntry'
import { LogSystemConfig } from '../../models/entities/LogConfig'

interface LogQueryState {
  logs: LogEntry[]
  filteredLogs: LogEntry[]
  isLoading: boolean
  error: string | null
  filter: LogQueryFilter
  totalCount: number
  page: number
  pageSize: number
}

export class LogQueryViewModel extends BaseViewModel<LogQueryState> {
  private liveLoggingUnsubscribe?: () => void

  constructor() {
    const initialState: LogQueryState = {
      logs: [],
      filteredLogs: [],
      isLoading: false,
      error: null,
      filter: {
        startTime: new Date(Date.now() - 24 * 60 * 60 * 1000), // 默认查询最近24小时
        endTime: new Date(),
        levels: Object.values(LogLevel),
        sources: Object.values(LogSource),
        modules: [],
        keywords: [],
        limit: 100,
        offset: 0
      },
      totalCount: 0,
      page: 1,
      pageSize: 100
    }
  
    super(initialState)
  }

  protected async onInitialize(): Promise<void> {
    await this.loadLogs()
    await this.startLiveLogging()
  }

  protected async onDispose(): Promise<void> {
    if (this.liveLoggingUnsubscribe) {
      this.liveLoggingUnsubscribe()
    }
  }

  // 公共方法
  public async refresh(): Promise<void> {
    await this.withLoading(async () => {
      await this.loadLogs()
    })
  }

  public async updateFilter(newFilter: Partial<LogQueryFilter>): Promise<void> {
    this.state.filter = {
      ...this.state.filter,
      ...newFilter,
      limit: this.state.pageSize,
      offset: (this.state.page - 1) * this.state.pageSize
    }
  
    await this.loadLogs()
  }

  public async changePage(newPage: number): Promise<void> {
    this.state.page = newPage
    this.state.filter.offset = (newPage - 1) * this.state.pageSize
  
    await this.loadLogs()
  }

  public async changePageSize(newSize: number): Promise<void> {
    this.state.pageSize = newSize
    this.state.page = 1
    this.state.filter.limit = newSize
    this.state.filter.offset = 0
  
    await this.loadLogs()
  }

  public async exportLogs(format: 'json' | 'csv' | 'text', includeDetails: boolean = false): Promise<string> {
    try {
      this.state.isLoading = true
      const result = await loggingService.exportLogs(this.state.filter, format, includeDetails)
      return result
    } catch (error) {
      this.state.error = error instanceof Error ? error.message : '导出失败'
      throw error
    } finally {
      this.state.isLoading = false
    }
  }

  public clearFilter(): void {
    this.state.filter = {
      startTime: new Date(Date.now() - 24 * 60 * 60 * 1000),
      endTime: new Date(),
      levels: Object.values(LogLevel),
      sources: Object.values(LogSource),
      modules: [],
      keywords: [],
      limit: this.state.pageSize,
      offset: 0
    }
    this.state.page = 1
  }

  // 私有方法
  private async loadLogs(): Promise<void> {
    try {
      this.state.isLoading = true
      this.state.error = null
    
      const logs = await loggingService.queryLogs(this.state.filter)
      this.state.logs = logs
      this.state.filteredLogs = this.applyLocalFilters(logs)
      this.state.totalCount = logs.length // 注意：这里应该是总记录数，需要后端支持
    } catch (error) {
      this.state.error = error instanceof Error ? error.message : '加载日志失败'
      console.error('Failed to load logs:', error)
    } finally {
      this.state.isLoading = false
    }
  }

  private async startLiveLogging(): Promise<void> {
    try {
      this.liveLoggingUnsubscribe = await loggingService.startLiveLogging((log) => {
        // 添加新日志到列表
        this.state.logs.unshift(log)
      
        // 应用过滤器
        if (this.matchesFilter(log)) {
          this.state.filteredLogs.unshift(log)
        }
      
        // 保持列表长度
        if (this.state.logs.length > 1000) {
          this.state.logs.pop()
        }
        if (this.state.filteredLogs.length > 1000) {
          this.state.filteredLogs.pop()
        }
      })
    } catch (error) {
      console.error('Failed to start live logging:', error)
    }
  }

  private applyLocalFilters(logs: LogEntry[]): LogEntry[] {
    return logs.filter(log => this.matchesFilter(log))
  }

  private matchesFilter(log: LogEntry): boolean {
    // 时间过滤
    if (this.state.filter.startTime && log.timestamp < this.state.filter.startTime) {
      return false
    }
    if (this.state.filter.endTime && log.timestamp > this.state.filter.endTime) {
      return false
    }
  
    // 级别过滤
    if (this.state.filter.levels && this.state.filter.levels.length > 0) {
      if (!this.state.filter.levels.includes(log.level)) {
        return false
      }
    }
  
    // 来源过滤
    if (this.state.filter.sources && this.state.filter.sources.length > 0) {
      if (!this.state.filter.sources.includes(log.source)) {
        return false
      }
    }
  
    // 模块过滤
    if (this.state.filter.modules && this.state.filter.modules.length > 0) {
      if (!this.state.filter.modules.includes(log.module)) {
        return false
      }
    }
  
    // 关键词过滤
    if (this.state.filter.keywords && this.state.filter.keywords.length > 0) {
      const logText = `${log.message} ${log.module} ${log.source}`.toLowerCase()
      const hasKeyword = this.state.filter.keywords.some(keyword =>
        logText.includes(keyword.toLowerCase())
      )
      if (!hasKeyword) {
        return false
      }
    }
  
    return true
  }

  // 计算属性
  public get levelCounts(): Record<LogLevel, number> {
    const counts: Record<string, number> = {}
  
    Object.values(LogLevel).forEach(level => {
      counts[level] = this.state.filteredLogs.filter(log => log.level === level).length
    })
  
    return counts as Record<LogLevel, number>
  }

  public get sourceCounts(): Record<LogSource, number> {
    const counts: Record<string, number> = {}
  
    Object.values(LogSource).forEach(source => {
      counts[source] = this.state.filteredLogs.filter(log => log.source === source).length
    })
  
    return counts as Record<LogSource, number>
  }

  public get totalPages(): number {
    return Math.ceil(this.state.totalCount / this.state.pageSize)
  }

  public get hasPrevPage(): boolean {
    return this.state.page > 1
  }

  public get hasNextPage(): boolean {
    return this.state.page < this.totalPages
  }
}

  public async loadConfig(): Promise<void> {
    try {
      this.state.isLoading = true
      this.state.error = null
    
      const config = await loggingService.getConfig()
      this.state.config = config
      this.state.originalConfig = JSON.parse(JSON.stringify(config))
      this.state.isDirty = false
    } catch (error) {
      this.state.error = error instanceof Error ? error.message : '加载配置失败'

      this.state.isLoading = false
    }
  }

  public async saveConfig(): Promise<boolean> {
    try {
      this.state.isLoading = true
      this.state.error = null
    
      await loggingService.updateConfig(this.state.config)
      this.state.originalConfig = JSON.parse(JSON.stringify(this.state.config))
      this.state.isDirty = false
    
      return true
    } catch (error) {
      this.state.error = error instanceof Error ? error.message : '保存配置失败'
      console.error('Failed to save log config:', error)
      return false
    } finally {
      this.state.isLoading = false
    }
  }

    this.state.config = JSON.parse(JSON.stringify(this.state.originalConfig))
    this.state.isDirty = false
  }

    this.state.config = {
      ...this.state.config,
      ...newConfig,
      rotation: {
        ...this.state.config.rotation,
        ...(newConfig.rotation || {})
      },
      export: {
        ...this.state.config.export,
        ...(newConfig.export || {})
      },
      sources: {
        ...this.state.config.sources,
        ...(newConfig.sources || {})
      }
    }
    this.state.isDirty = true
  }

    this.updateConfig({
      sources: {
        ...this.state.config.sources,
        [source]: level
      }
    })
  }

    this.updateConfig({ mode })
  }

  public setDefaultLevel(level: LogLevel): void {
    this.updateConfig({ defaultLevel: level })
  }

  public toggleConsoleOutput(): void {
    this.updateConfig({ consoleOutput: !this.state.config.consoleOutput })
  }

  public toggleFileOutput(): void {
    this.updateConfig({ fileOutput: !this.state.config.fileOutput })
  }

  public toggleRotation(): void {
    this.updateConfig({
      rotation: {
        ...this.state.config.rotation,
        enabled: !this.state.config.rotation.enabled
      }
    })
  }

  public setMaxSize(mb: number): void {
    this.updateConfig({
      rotation: {
        ...this.state.config.rotation,
        maxSize: mb * 1024 * 1024
      }
    })
  }

  public setMaxFiles(count: number): void {
    this.updateConfig({

        ...this.state.config.rotation,
        maxFiles: count
      }
    })
  }

    })
  }

        compression: !this.state.config.rotation.compression
      }
    })
  }

  public setCompressionLevel(level: number): void {

        ...this.state.config.rotation,
        compressionLevel: Math.max(1, Math.min(9, level))
      }
    })
  }

  public get modeOptions() {
    return Object.values(LogMode).map(mode => ({
      label: mode === LogMode.DEVELOPMENT ? '开发模式' : 
             mode === LogMode.PRODUCTION ? '生产模式' : '测试模式',
      value: mode
    }))
  }

  public get levelOptions() {
    return Object.values(LogLevel).map(level => ({
      label: level,
      value: level
    }))
  }

  public get sourceOptions() {
    return Object.values(LogSource).map(source => ({
      label: this.formatSourceName(source),
      value: source
    }))
  }

  }

      [LogSource.UI]: '用户界面',
      [LogSource.UPDATE]: '更新',
      [LogSource.PLUGIN]: '插件'
    }
    return names[source] || source
  }
}

const getSourceColor = (source: string): string => {
  // 简单的颜色映射
  const colors = [

    '#118ab2', '#ef476f', '#073b4c', '#7209b7', '#f72585'
  ]

  let hash = 0
  for (let i = 0; i < source.length; i++) {
    hash = source.charCodeAt(i) + ((hash << 5) - hash)
  }

  return colors[Math.abs(hash) % colors.length]
}

// src/core/logging/init.ts
import { loggingService, logger } from '../../models/services/LoggingService'
import { LogSource } from '../../models/entities/LogEntry'

export async function initLogging(): Promise<void> {
  try {
    // 加载日志配置
    await loggingService.getConfig()
  
    // 记录应用启动日志
    logger.info(LogSource.SYSTEM, 'application', '应用程序启动')
  
    // 设置全局错误处理
    setupGlobalErrorHandling()
  
    console.log('Logging system initialized successfully')
  } catch (error) {
    console.error('Failed to initialize logging system:', error)
    throw error
  }
}

function setupGlobalErrorHandling(): void {
  // 全局错误捕获
  window.addEventListener('error', (event) => {
    logger.error(
      LogSource.SYSTEM,
      'global',
      `未捕获的错误: ${event.message}`,
      {
        filename: event.filename,
        lineno: event.lineno,
        colno: event.colno
      },
      event.error
    )
  })

  // Promise拒绝捕获
  window.addEventListener('unhandledrejection', (event) => {
    logger.error(
      LogSource.SYSTEM,
      'global',
      '未处理的Promise拒绝',
      { reason: event.reason?.toString() },
      event.reason
    )
  })

  // Vue错误处理
  const app = (window as any).__VUE_APP__ // 假设有全局Vue应用引用
  if (app?.config?.errorHandler) {
    const originalErrorHandler = app.config.errorHandler
    app.config.errorHandler = (err: Error, instance: any, info: string) => {
      logger.error(
        LogSource.UI,
        'vue',
        `Vue组件错误: ${info}`,
        {
          component: instance?.$options?.name,
          info
        },
        err
      )
    
      // 调用原始错误处理器
      if (originalErrorHandler) {
        originalErrorHandler(err, instance, info)
      }
    }
  }
}
